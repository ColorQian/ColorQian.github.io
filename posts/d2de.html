单向环形链表之约瑟夫环

约瑟夫环
数据结构
发布日期:   2021-08-07
文章字数:   690
阅读时长:   3 分

  • Josephus问题:

    约瑟夫问题为：设编号为1，2，… n的n个人围坐一圈，约定编号为k（1<=k<=n）的人从1开始报数，数到m 的那个人出列，它的下一位又从1开始报数，数到m的那个人又出列，依次类推，直到所有人出列为止，由此产生一个出队编号的序列。

    例如：

    n = 5 , 即有5个人

    k = 1, 从第一个人开始报数

    m = 2, 数2下

    出圈的顺序

    2->4->1->5->3

下面是单向环形链表类

//单向环形链表-约瑟夫环
public class SingleCircle &#123;


    private Boy first;

    /**
     * 将小孩加入约瑟夫环的方法
     * @param num 表示环中结点的个数
     */
    public void add(int num)&#123;
        first = null;

        Boy curBoy = null;  //辅助变量，用来帮助小孩进圈
        for (int i = 1; i <= num; i++) &#123;
            Boy boy = new Boy(i);

            //当加入第一个小孩时，让其自己成环
            if (i == 1)&#123;
                first = boy;  //first始终指向第一个小孩
                first.next = first;
                curBoy = first;
            &#125;

            //新的小孩结点的进圈操作
            curBoy.next = boy;
            curBoy = curBoy.next;
            curBoy.next = first;

        &#125;
    &#125;

    //遍历圈中的小孩结点
    public void list()&#123;

        if (first == null)&#123;
            System.out.println("圈中没有小孩结点");
            return;
        &#125;

        Boy cur = first;
        while (true)&#123;
            System.out.printf("当前的小孩编号:%d \n",cur.no);
            if (cur.next == first)&#123;
                break;
            &#125;
            cur = cur.next;
        &#125;
    &#125;


    /**
     * 小孩出圈的方法
     * @param startNo 从第几个小孩开始数
     * @param count 数几次
     * @param num 圈中小孩的个数
     */
    public void remove(int startNo,int count,int num)&#123;
        if (first == null || startNo < 1 || startNo > num)&#123;
            System.out.println("参数输入有误！！！");
            return;
        &#125;

        Boy helper = first;//创建一个辅助指针，让其始终位于first指针的前一个

        //先让first移动到startNo处，即开始报数的位置
        for (int i = 1; i <= startNo-1; i++) &#123;
            first = first.next;
        &#125;

        //再让helper 移动到first的前一位
        while (helper.next != first)&#123;
            helper = helper.next;
        &#125;

        while (helper != first)&#123;

            //开始报数，让helper和first移动到指定位置
            for (int i = 1; i <= count-1; i++) &#123;
                helper = helper.next;
                first = first.next;
            &#125;

            //此时first指向要移除的小孩结点
            System.out.printf("出圈小孩的编号：%d \n",first.no);
            first = first.next;
            helper.next = first;
        &#125;

        System.out.printf("最后出圈小孩的编号：%d \n",first.no);
    &#125;


    //***************写个main方法测试一下*******************

    public static void main(String[] args) &#123;
        SingleCircle singleCircle = new SingleCircle();

        singleCircle.add(5);
        
        singleCircle.remove(1,2,5);
    &#125;
